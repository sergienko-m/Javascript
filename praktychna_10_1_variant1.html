Promise в JavaScript
// Об'єкт що представляє майбутній результат асинхронної операції

Стани Promise

⏳ pending Початковий стан — операція ще виконується
✅ fulfilled Операція завершилась успішно — є результат
❌ rejected Операція завершилась з помилкою

Синтаксис

// Створення Promise
const myPromise = new Promise((resolve, reject) => {
  // асинхронна операція
  setTimeout(() => {
    if (success) resolve({ data: 'результат' });
    else         reject(new Error('помилка'));
  }, 1000);
});

// Використання через .then() .catch() .finally()
myPromise
  .then(result  => console.log(result))
  .catch(error  => console.error(error))
  .finally(()  => console.log('завжди виконується'));

Promise chains — .then()
// Ланцюжок промісів — кожен .then() отримує результат попереднього
checkAvailability(orderId)
  .then(result => {
    // result — дані від checkAvailability
    return reserveItems(result.orderId);
  })
  .then(result => {
    // result — дані від reserveItems
    return processPayment(result.orderId, result.amount);
  })
  .then(result => {
    return scheduleDelivery(result.orderId);
  })
  .catch(error => {
    // Спрацює при будь-якій помилці в ланцюжку
    console.error('Помилка:', error.message);
  })
  .finally(() => {
    // Виконається завжди — успіх або помилка
    cleanup();
  });

step1()

    handleError(err);
  });

Відновлення після помилки

// .catch() може повернути значення
step1()
  .then(step2)
  .catch(err => {
    return { fallback: true }; // відновлення
  })
  .then(result => {
    // result.fallback === true
  });

Promise.all
// Паралельне виконання — чекає на ВСІ проміси
// Всі запити виконуються ПАРАЛЕЛЬНО
Promise.all([
  fetchMenu(),         // 1с
  fetchRestaurants(), // 1.5с
  fetchReviews()      // 0.8с
]).then(([menu, restaurants, reviews]) => {
  // Всі результати одразу — через 1.5с (найдовший)
});
// Якщо хоч один rejected — весь Promise.all rejected

Promise.race
// Повертає результат ПЕРШОГО завершеного промісу
// Виграє найшвидший
Promise.race([
  fetchFromServer1(), // ~1.2с
  fetchFromServer2(), // ~0.8с ← виграє
  fetchFromServer3(), // ~1.5с
  new Promise((_, reject) =>
    setTimeout(() => reject(new Error('Timeout!')), 3000)
  ) // таймаут 3с
]);

Promise.allSettled
// Чекає на ВСІ проміси незалежно від результату
// На відміну від Promise.all — не зупиняється при помилці
Promise.allSettled([
  fetchMenu(),       // може провалитись
  fetchReviews(),    // може провалитись
  fetchPromos()      // може провалитись
]).then(results => {
  results.forEach(result => {
    if (result.status === 'fulfilled')
      console.log(result.value);
    else
      console.error(result.reason);
  });
});

  console.log('3 — macrotask'), 0);    // виконається 3-м

  .then(() =>
    console.log('2 — microtask'));       // виконається 2-м
